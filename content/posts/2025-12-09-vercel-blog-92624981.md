Vercel 现已支持 FastAPI 应用的生存周期事件。这允许您定义可在启动和优雅关闭时执行的逻辑——例如管理数据库连接或刷新外部日志。
from contextlib import asynccontextmanagerfrom fastapi import FastAPI
@asynccontextmanagerasync def lifespan(app: FastAPI): # 启动逻辑 print("Starting up...") await startup_tasks() yield # 关闭逻辑 await cleanup_tasks()
app = FastAPI(lifespan=lifespan)
